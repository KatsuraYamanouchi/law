import asyncio
import re

from typing import AsyncIterable, AsyncIterator
#from fastapi import FastAPI

async def combine_streams(
        *streams: AsyncIterator[fp.PartialResponse]
) -> AsyncIterator[fp.PartialResponse]:
    """Combines a list of streams into one single response stream.
    Allows you to render multiple responses in parallel.
    """
    active_streams = {id(stream): stream for stream in streams}
    responses: dict[int, list[str]] = defaultdict(list)

    async def _advance_stream(
        stream_id: int, gen: AsyncIterator[fp.PartialResponse]
    ) -> tuple[int, fp.PartialResponse | None]:
        try:
            return stream_id, await gen.__anext__()
        except StopAsyncIteration:
            return stream_id, None

    while active_streams:
        for coro in asyncio.as_completed(
            [
                _advance_stream(stream_id, gen)
                for stream_id, gen in active_streams.items()
            ]
        ):
            stream_id, msg = await coro
            if msg is None:
                del active_streams[stream_id]
                continue

            if isinstance(msg, fp.MetaResponse):
                continue
            elif msg.is_suggested_reply:
                yield msg
                continue
            elif msg.is_replace_response:
                responses[stream_id] = [msg.text]
            else:
                responses[stream_id].append(msg.text)

            text = "\n\n".join(
                "".join(chunks) for stream_id, chunks in responses.items()
            )
            yield fp.PartialResponse(text=text, is_replace_response=True)

def preprocess_message(message: fp.ProtocolMessage, bot: str) -> fp.ProtocolMessage:
    """Process bot responses to keep only the parts that come from the given bot."""
    if message.role == "bot":
        parts = re.split(r"\*\*([A-Za-z_\-\d]+)\*\* says:\n", message.content)
        for message_bot, text in zip(parts[1::2], parts[2::2]):
            if message_bot.casefold() == bot.casefold():
                return message.model_copy(update={"content": text})
        # If we can't find a message by this bot, just return the original message
        return message
    else:
        return message

def preprocess_query(request: fp.QueryRequest, bot: str) -> fp.QueryRequest:
    """Parses the two bot responses and keeps the one for the current bot."""
    new_query = request.model_copy(
        update={
            "query": [preprocess_message(message, bot) for message in request.query]
        }
    )
    return new_query

async def stream_request_wrapper(
    request: fp.QueryRequest, bot: str
) -> AsyncIterator[fp.PartialResponse]:
    """Wraps stream_request and labels the bot response with the bot name."""
    label = fp.PartialResponse(
        text=f"**{bot.title()}** says:\n", is_replace_response=True
    )
    yield label
    async for msg in fp.stream_request(
        preprocess_query(request, bot), bot, request.access_key
    ):
        if isinstance(msg, Exception):
            yield fp.PartialResponse(
                text=f"**{bot.title()}** ran into an error", is_replace_response=True
            )
            return
        elif msg.is_replace_response:
            yield label
        # Force replace response to False since we are already explicitly handling that case above.
        yield msg.model_copy(update={"is_replace_response": False})

..バイスティックの７原則
対象者をクライエント、援助者をワーカーという
1. 個別化の原則
　クライエントの抱える困難や問題は、どれだけ似たようなものであっても、人それぞれの問題であり「同じ問題は存在しない」とする考え方。この原則において、クライエントのラベリング（人格や環境の決めつけ）やカテゴライズ（同様の問題をまとめて分類してしまい、同様の解決手法を執ろうとする事）は厳禁となる。
※自己点検：外面的に同じような状況に見えても、人それぞれ育ってきた環境が違い、価値観も違う。今目の前にいる人は世界に一人しかいないということを肝に銘じておかなければならない。援助がパターン化していないか。偏見や先入観にとらわれていないか。自分のペースで話を進めていないか。「忙しそうですね」と言われていないか。

2. 意図的な感情表現の原則
　クライエントの感情表現の自由を認める考え方。特に抑圧されやすい否定的な感情や独善的な感情などを表出させることでクライエント自身の心の 枷かせを取り払い、逆にクライエント自身が自らを取り巻く外的・内心的状況を俯瞰しやすくする事が目的。またワーカーもクライエントに対しそれが出来るように、自らの感情表現を工夫する必要がある。
※自己点検：話しやすい雰囲気を意識しているか。座る位置はそこで良いのか。開かれた質問と閉じられた質問を意識しているか。リラックスできているか。同じ流れの中に入れているか。

3. 統制された情緒関与の原則
　ワーカー自身がクライエント自身の感情に呑み込まれないようにする考え方。クライエントを正確にかつ問題無くケース解決に導くため「ワーカー自身がクライエントの心を理解し、自らの感情を統制して接していく事」を要求する考え方。
※自己点検：自分の感情を自覚できているか。今抱いている感情は誰の感情なのか。共感の及ぼす過度な感情移入をしていないか。目的を意識しながら反応できているか。時期は適切か。急ぎすぎていないか。安易な情緒的関与をしていないか。平常心は保てているか。

4. 受容の原則
　クライエントの考えは、そのクライエントの人生経験や必死の思考から来るものであり、クライエント自身の『個性』であるため「決して頭から否定せず、どうしてそういう考え方になるかを理解する」という考え方。この原則によってワーカーによるクライエントへの直接的命令や行動感情の否定が禁じられる。
※ 自己点検：クライエントの人となりを吟味しているか。今起きている現実をありのまま受け止められているか。ギアはニュートラルに入っているか。ハンドルにあそびはあるか。

5. 非審判的態度の原則
　クライエントの行動や思考に対して「ワーカーは善悪を判じない」とする考え方。あくまでもワーカーは補佐であり、現実にはクライエント自身が自らのケースを解決せねばならないため、その善悪の判断もクライエント自身が行うのが理想とされる。また人間は基本的に当初において自らを否定するものは信用しないため受容の観点からも、これが要求される。
※自己点検：違う角度からもみるようにしているか。多面的に捉えているか。色目めがねをかけていないか。木も森もみえているか。常識という枠にとらわれていないか。

6. 自己決定の原則
　「あくまでも自らの行動を決定するのはクライエント自身である」とする考え方。問題に対する解決の主体はクライエントであり、この事によってクライエントの成長と今後起こりうる同様のケースにおけるクライエント一人での解決を目指す。この原則によって、ワーカーによるクライエントへの命令的指示が否定される。
※自己点検：今目の前にいる人が本来持っている生きる力や強さはどのくらいあるのか。本人の意思をしっかり確認しているか。実際発した言葉は本心なのか。緊急性はあるのか。周りの人の援助は期待できるか。援助が行き過ぎていないか。働きかけによって解決できる力が発揮できる可能性があるか。

7. 秘密保持の原則
　クライエントの個人的情報・プライバシーは絶対に他方にもらしてはならないとする考え方。いわゆる「個人情報保護」の原則。他方に漏れた情報が使われ方によってクライエントに害を成す可能性があるため。
※自己点検：秘密保持を意識しているか。個人情報の使用にあたってクライエントに不安をあたえていないか。個人情報の管理はしっかりできているか。

　　突然の来訪は、予約をしてもらわないとお会いできないことを伝えて、待ってもらわずにいったんお引き取り願い，こちらから電話する旨伝える
　　弁護士から事前に指示を受けていない限り，金品を預からないようにすること
　　ただし，緊急の要件（前記参照）である場合には，念のため弁護士の携帯電話へ連絡して指示を仰ぐ
　３　顧客や相手方以外の第三者，営業など
　　営業に関しては応答不要
　　それ以外の不明来訪者は，差し支えなければ用件まで伺っておき，伝言メモのみでよい
特殊ケース＞
　弁護士が相談予約を忘れていた場合
　　＊直ちに弁護士に連絡をとり，出来ればその場で打ち合わせて新しい期日を入れる。
　　　詫びとして３０００円程度（遠方来訪の場合は交通費を勘案して適宜加算）の商品券を渡す
　弁護士が在室していないことになっている場合

..経理処理規則
　弁護士報酬等請求権の時効は主観５年、客観１０年
　現金主義経理を原則とする。

着手金
　着手金請求権発生日は原則として請求書発行日または委任状作成日とする
　委任状作成日が不明の場合には，初回相談，受任通知，提訴等のいずれかをもって充てる

報酬金
　報酬金請求権発生日は顧客の利得が、示談書、差押え等で現実に確定し、顧客に交付する準備が完了した時とする
　債務者・被告事件の場合には棄却・和解等の確定時に利得があったものと看做す
　未受領分は売掛として計上し，後日現金受領時に振り替え処理する

裁判所費用
　印紙代・予納金・保管金は依頼者の負担とし、必要に応じて事前に預り金として受領し、そこから支出する
　予納郵券は原則として弁護士の負担とし，着手金の規模に比べて高額にならない限り、依頼者負担にはしない
　預り金がない場合の上記の支出については，弁護士と協議して一般事務費から支出する

経費
　交通費・通信費・振込手数料等の事務手続費用は原則として弁護士負担とする
　ただし，遠方の事件や相手方多数などの特殊事例については依頼者との協議により，相当の実費を預かる場合があるので，その場合には別途預り金管理する
　消費税を明示計上していないときは、内税を算出して必ず計上する

中途解約時の控除・清算基準
　着手前：原則全額返還、ただし相談時間が１時間を超える分につき１時間あたりタイムチャージ１１，０００円（内税１，０００円）を控除
　訴状等を裁判所に提出したときより後：６割を控除
　相手方が応訴後尋問開始前に中断した場合：８割を控除
　尋問開始または和解協議後：返還なし

Today’s financial advice: Just say no.
You can probably think of instances when an individual ought to ignore one or two of the suggestions below. Still, I’d argue that, if most folks followed these rules, they’d be far better off financially. Want a brighter financial future? Here are 51 things you shouldn’t do:

1. Don’t buy cash-value life insurance.
2. Don’t envy hedge fund investors.
3. Don’t write frequent checks against bond funds held in a taxable account.
4. Don’t carry a credit card balance.
5. Don’t invest in high-turnover stock funds.
6. Don’t fund custodial accounts if your family hopes to receive college financial aid.
7. Don’t trust brokers when their lips are moving.
8. Don’t keep a heap of money in your checking account.
9. Don’t assume the premium on your long-term-care insurance is fixed for life.
10. Don’t forget that a high potential return means high risk.
11. Don’t buy a home if you think you’ll move in the next five years.
12. Don’t invest 100% in stocks — or 100% in bonds.
13. Don’t die without a will.
14. Don’t buy trip-cancellation insurance.
15. Don’t retire with debt.
16. Don’t buy initial public stock offerings.
17. Don’t throw away the advantages of index funds by actively trading them.
18. Don’t claim Social Security at age 62.
19. Don’t buy a tax-deferred annuity in an individual retirement account.
20. Don’t apply for credit too often.
21. Don’t use your taxable account to buy high-yield junk bonds or real estate investment trusts (REITs).
22. Don’t opt for low insurance deductibles.
23. Don’t fully fund your 401(k) if you smoke, drink heavily and never exercise.
24. Don’t buy any fund with annual expenses above 0.35%.
25. Don’t instinctively hang on to losing stocks.
26. Don’t be surprised if every solution offered by an insurance salesman involves insurance.
27. Don’t assume you, or anybody else, are smarter than the market.
28. Don’t get your stock picks from your brother-in-law, your spam folder or the television.
29. Don’t purchase life insurance if you don’t have financial dependents.
30. Don’t pay a 6% real estate commission.
31. Don’t opt for the extended warranty.
32. Don’t invest heavily in your employer’s stock.
33. Don’t purchase a house that’s bigger than you really need.
34. Don’t day trade.
35. Don’t have children if you hope to retire early.
36. Don’t read anything into short-term market movements.
37. Don’t buy investments without first settling on your financial goals.
38. Don’t use more than 10% of the credit limit on your credit cards.
39. Don’t buy an individual bond without figuring out what markup you’re paying.
40. Don’t forget about inflation.
41. Don’t buy an investment unless you’d be happy to hold it for 10 years.
42. Don’t assume a commission-free stock trade is cost-free.
43. Don’t buy based on past performance and expect it to persist.
44. Don’t leave your ex-spouse listed as your 401(k) plan’s beneficiary.
45. Don’t take out a large mortgage just for the tax deduction.
46. Don’t keep money in the stock market that you’ll need to spend within five years.
47. Don’t buy variable annuities.
48. Don’t assume a high yield means a high return.

50. Don’t expect stocks to earn 10% a year, even over the long run.
51. Don’t lend money to family members if you’ll need it back.

事件の見通しと費用の説明
　時間がかかっても，はじめから終わりまで一気に説明してしまう
　中途半端に途中まで説明しただけでは依頼者は極度の不安に陥りがち
　説明内容は記録して後日再現できるように
　依頼者の主張・意見に惑わされずに，事実と証拠から主張を組み立てる
　理解力の劣る場合や判断能力に不安のある依頼者等への説明は必ず録音・録画をとっておく
　可能な限り相手の目の前で成果を出すことが信頼へつながる
　受任の範囲を明らかにする。将来の展開でかかる追加費用の見込みも説明する
　提訴の前には関係条文，判例を可能な限り全部検索して，依頼者へ出来る限り全体像・見通しを説明する

依頼者の正当な利益
　何事もすべて依頼者の同意・承認に基づくこと（ベストインタレストでも同意・承認されるとは限らないが、あくまでも依頼者を主役とする）
　法律違反はもちろんのこと，道徳・倫理違反の指導もすべきでない
　依頼者から整理を一任された場合でも，相手方の信用を利用して虚偽の事実を告げて弁済猶予を求めたりしない
　売主代理人として立ち会う場合は必要書類や現況調査を怠らない

交渉一般
　原理原則を最優先させること（安易な妥協はしない
　関係者との連絡（郵便，電話，ファックス，会話）はきちんと記録を残す
　もめそうな相手との会話は必ず録音する
　依頼者の意思確認は幾重にも慎重に行う
　結論を急ぐときも急がないときも慎重に判断する
　弁護士のイメージを良くも悪くも利用せよ。時には役者のように演じることも必要
　依頼者との信頼関係を重視する
　相手を自分のペースに乗せる
　相手に恩を売っても，決して借りを作らない
　安易な信用は禁物（占有移転禁止仮処分は有効
　仮処分の空振りをおそれない（何度でも執拗に
　仮差押・仮処分は原則どおりに粛々と
　封印破棄しないよう気をつける

示談・和解
　できないことは当初から明確にしておく
　和解期日に相手方弁護士に対し保証金還付金をもってする和解金の全額一括支払いを口頭で約束したにも関わらず，還付金を依頼者に全額交付したため一部しか支払いがなされない事態を招くことがあるので，注意（依頼者であっても過度の信頼は不可）
　先に自分の結論を出しすぎると損
　和解の内容は細部まで主導権を握る（相手まかせはよくない）
　和解時の当事者の表示には気を配る（送達不能や受領拒否につながる
　和解条項はきちんと吟味
　当事者複数の訴訟における一部当事者間の和解は注意する

訴訟
　請求の趣旨の再検討（執行がスムーズにできるのかどうか
　　特に登記請求は司法書士・管轄法務局の事前承認を得ておく
　請求債権目録のチェック
　特定が重要（期間、時期等）
　メインの争点との思い込みに引きずられないように、最初は広く浅く主張反論しておく
　同姓同名の他人でないかどうかを確認（住民票以外の情報との照合を要する）
　訴訟費用と控訴費用を説明し，どこまでいくか事前に打ち合わせておく
　文書は１Ｗ前までに必ず提出する
　些細な反対証拠にも反証を忘れない（なめていると痛いことになる
　出廷に当たっての心構え（記録をよく読む
　わからないことは恥を承知で裁判所に聞く
　依頼者のいうことを鵜呑みにしない
　裁判官の常識が世間の常識と同一であると限らない

証人尋問
　速記・録音が取りやすいように、ゆっくり大きな声で尋問する
　何を立証するかを常に考えながら尋問すること
　固有名詞のメモは，事前に書記官に提出しておく
　時間厳守のためには十分な準備が必要
　不利供述は時間を越えてでも無効化すること
　反復質問は裁判官へ印象づけなければならない場面に限ること
　答と質問が重ならないように注意する（録音できない

トラブル対応
　是々非々の態度で臨み，辞任を躊躇しない
　途中辞任の場合には懲戒請求に備えて、記録を整備し、辞任経過を明らかにして，仕事内容を弁護士会に開示できるようにしておく
　不利事実・不利益文書の秘匿をしない

他の弁護士の受任事件への対応
　　弁護士は相互に名誉と信義を重んじ、みだりにほかの弁護士を誹謗中傷してはならない
　　弁護士は正当な職務慣行または信義に反して、ほかの弁護士を不利益に陥れてはならない
　相手方弁護士の情報・意見を依頼者に告知する限度については，慎重な判断を要する
　本人同士が直接交渉する場合の助言は，弁護士がつくかどうかで変化させる
　　＊本人当事者の理解力には限界があるので，理屈はかえって逆効果

　 修正で済むときでも，可能な限り作り直しを求める
　 作り直しの書類にも代筆ではなく本人の自署を求める

..住所等の秘匿について 大阪家庭裁判所家事第４部 人事訴訟係
人事訴訟は，家事手続と異なり，訴訟記録は閲覧謄写することができるのが原則です（民訴９１条）。単に相手方に住所等を知られたくないというだけでは秘匿は認められませんし，仮に裁判所が秘匿を認めたとしても，それは，当事者に対して，住所等，本来記載すべき情報を訴状等に記載しないことを許容し，裁判所が作成する文書に秘匿情報を記載しない扱いをすることであり，既に記録にあらわれた秘匿情報を裁判所が自らマスキングすることはできません。したがって，秘匿を希望する当事者が書類等を提出するときは，提出書類等に秘匿情報（住所やマイナンバー等）があらわれないように当事者自身が十分に注意し，確認してください。
特に委任状，書証（診断書，源泉徴収票，未成年子の学校関係資料など），訴訟救助の疎明資料，年金分割のための情報通知書等には，住所等が記載されていることが多いことから事前に十分な確認が必要です。
 なお，訴えの提起に当たり住所等の秘匿を希望するときは，住所の表示は「大阪府（以下秘匿）」とし，秘匿を希望する旨及びその必要性を具体的に記載した文書を提出してください。

..保護命令申立書の使い方
１　本書式は，申立人に代理人弁護士が就き，代理人弁護士が申立てをする場合の書式となっております。申立人本人が申立てをする場合の書式は，次のとおり大阪地方裁判所第１民事部のホームページからダウンロードできますので，それをご利用願います。http://www.courts.go.jp/osaka/vcms_lf/05.hogomeirei.pdf
２　本書式の，ハイライト表示されている部分は，文字やチェックを入れることができるようになっておりますので，申立ての内容にそって，必要な部分を入力してください。入力する文字数には制限がありますので，申立書はなるべく簡潔に事実を記載するようにお願いします。記載しきれない場合は，適宜補充用紙に記載し，それを申立書に引用していただいて構いません。
３　利用するパソコンの環境によっては，ハイライト部分の表示が見えにくい場合もあります。その場合，アドビリーダーの「編集」→「環境設定」を開き，分類（G）から「フォーム」を選択して，「フィールドのハイライトの表示」の色を見えやすい色に変更してください。
４　利用するパソコンのOSがウインドウズVISTA以前である場合，あるいはアドビリーダーが最新のヴァージョンに更新されていない場合は，フォームに入力したデータが保存されない場合があります。その場合，データの保存はできませんので，入力したデータを印刷しておくなどしておいてください。
５　本書式を利用して，申立てを行う場合は，申立書一式（3頁以降）をそのまま印刷して，提出してください。申立てに関わらない部分についても省略しないようにご注意願います。
６　本書式について，要望や意見がございましても，個別には対応しかねます。当部と弁護士会との協議会等の機会を通じて，ご要望等は承りますので，ご協力願います。

第２ 相続人からの申請
下記の必要書類を同封の上，申請先の裁判所まで郵送してください。
１ 申請書
「相続放棄申述受理証明申請書（利害関係人用）」に必要事項を記入し，申請者の印（認め印可，スタンプ印不可）を押印してください。
※ 事件番号・受理年月日が不明の場合は，証明書の申請の前に，「相続放棄・限定承認の申述の有無についての照会」を行ってください。
２ 身分証明書（運転免許証，健康保険証等）のコピー
３ 利害関係疎明資料（いずれもコピーで可，原本の添付不要）
 一般的には以下の書類となりますが，相続関係により異なりますので，詳細につきましては,事前に申請先の裁判所までお電話にてお問い合わせください。
 ① 被相続人の死亡の記載のある戸籍(除籍)謄本
 ② 申請者の戸籍謄本
４ 手数料
 申述人 1 人の証明書１通につき，１５０円分の収入印紙

普通郵便の場合で受理証明書４通まで８２円分，５通以上９２円分
 【注意】
 すでに相続放棄済みの方に対しては，同じ被相続人について，相続放棄した他の親族の方の相続放棄申述受理証明書を発行することは原則としてできませんので，御了承ください。
※ 例えば，被相続人の子３人の相続放棄申述が受理された場合において，相続放棄済の１人の方から相続放棄をした他の２人分の受理証明書を一括して交付申請することは原則としてできません（それぞれ，御自身が御自身の分の証明書を交付申請いただく必要があります。）。
これは，相続放棄をした方は，その相続については，初めから相続人とならなかったものとみなされ（民法９３９条），相続人としての利害関係が認められなくなるからです。
※ ただし，相続関係以外の利害関係がある場合は発行することができる場合もあります。

.■小額事件受任に係る補助金制度について
基準＞
　着手金８％　報酬１６％

受領した分と５万円の差額のうち３万を補助
　例１　１０万もらうとき
　　差額は５万だが、３万の補助しか出ないので、顧客は実質７万の支払いで、弁護士は１０万の報酬になる。
　例２　２万もらうとき
    差額は３万、３万の補助がでるので、顧客は２万でいいが、弁護士には５万の報酬になる。
　例３　５万もらうとき
    差額がないので、補助はない。顧客は実質５万の支払い

Ｑ：預り金の清算不能の場合の請求について
  営業所運営費として、１０万円預かる。期末時又は退職時に清算することを確約する。清算不能の場合は、給与より天引きされても異議申立しない。
  勝手に退職されて、清算しないままに給料全額を支払った。どうやって、預り金を回収するか。
Ａ：法律問題
まず、本預り証の解釈としては、１０万円の消費寄託契約と解される（民６６６）。営業費用が会社負担であれば、受託者側では立替金処理になる。従って、預り証の清算条項の解釈としては、右立替金と寄託金との相殺を合意したものと見るのが論理的と思われる。→ 後の検討で単純な消費寄託でないことが分かる。
  本件では、寄託金返還期限が定められているので、その期限がくれば、原則として寄託金全額の返還を請求できる。だが、前記清算条項があるので、立替金及び寄託金を相殺処理した上でなければ、差額につき相互に支払を請求できないと考えるのが固い考え方であろう（もっとも、返還期限が７月３１日なのか、期末時なのか退職時なのかがはっきりしないという点、１０万円以上の営業経費を使ったときにスタッフサービス側に追加支払義務が生じるのか否かが書面上不明な点、どの範囲の営業経費までが含まれるのか不明である点、なども問題となる）。
  問題は、相手から立替金の請求が具体的になされない場合であるが、その場合は相殺すべき債務なしと考えて、全額の返還を求めるというのも合理的な考え方の一つであろうと思われる。

処理方針の問題
  内容証明が効かなければ、小額なので支払督促をかける。管轄は被告居住地が原則。合意があれば、ほかのところでも可能だが、それ以外の管轄の特例はないので、本件では難しいと思われる。
  相手は当然に立替金との相殺を主張するであろうから、結局は異議が出て訴訟ということになるだろうが、そうすると、費用対効果の関係で法的手段での回収は難しいのではないか。

結論
  よって、当面は全額返還の請求をしつこく進めるのがよいと思われる。

基本＞
第27条（解約による建物賃貸借の終了）
建物の賃貸人が賃貸借の解約の申入れをした場合においては、建物の賃貸借は、解約の申入れの日から六月を経過することによって終了する。
2 前条第二項及び第三項の規定は、建物の賃貸借が解約の申入れによって終了した場合に準用する。
→これが借家契約解約の場合の最低条件

第29条（建物賃貸借の期間）
期間を一年未満とする建物の賃貸借は、期間の定めがない建物の賃貸借とみなす。
→これにより1年未満の期間を定めても、6ヶ月の解約期間が必要になることになる。

第30条（強行規定）
この節の規定に反する特約で建物の賃借人に不利なものは、無効とする。
→この規定により、27条1項より短い解約期間を定めても無効になる
  結局、原則的には30日の解約期間は無効となる。


30条の例外となる場合＞
第38条（賃貸人の不在期間の建物賃貸借）
  転勤、療養、親族の介護その他のやむを得ない事情により、建物を一定の期間自己の生活の本拠として使用することが困難であり、かつ、その期間の経過後はその本拠として使用することとなることが明らかな場合において、建物の賃貸借をするときは、その一定の期間を確定して建物の賃貸借の期間とする場合に限り、第三十条の規定にかかわらず、契約の更新がないこととする旨を定めることができる。この場合には、第二十九条の規定を適用しない。
2 前項の特約は、同項のやむを得ない事情を記載した書面によってしなければならない。
→これは更新しないことを定めうるというだけの規定であり、しかも、
確定期間を定め、やむを得ない事情を記載した書面で特約をする必要がある。1年未満の期間を定めることが可能。

第40条（一時使用目的の建物の賃貸借）
  この章の規定は、一時使用のために建物の賃貸借をしたことが明らかな場合には、適用しない。
→これは目的自体が一時の使用でなければならない
賃料が相場より低い
賃料の改定が予定されていない
契約期間が短く定められている
賃借人がその物件に長居する必然性がない
  などの事情があれば、一時使用になりうるだろうが、実際の認定は裁判になってみないとわからないというところがある。

  以上のことから、安全策としては半年おくのが妥当ではあるが、契約が一時使用に準拠したものなら、多少の無理も通せる可能性はあるかも。

Ｑ：預かり証のチェック
  ２ヶ月間の営業経費として預ける。営業経費としての範囲では自由に使える。使った分は領収証等により営業経費として認めたものについては、返却不要。使わなかった分はいったん会社に返却する。返却すべき金を返さない場合、どのような請求ができるか。また、給与からの控除が可能か。
Ａ：
１  勤務というのは、雇用関係が発生していることが条件かどうか
２  単純な消費寄託ではない。預かりというより、むしろ金額範囲制限つき営業経費使用権原の付与（委任）ないし、労働契約上の経費取扱準則（無名契約）と解すべき。その意味ではタイトルが不適切ではないか。
３  「金額を・・・預かる。」というのは、そのとき現金を渡すのかそれとも、預金通帳などにしてあるのか不明。本来、本社経理部がすべき小口経費の管理事務を営業所長に任せる内容か。
４  預かり期限と清算期限との関係が不明瞭。1月31日を過ぎたら返却しなければならないのか。
５  実態をからすれば、問い合わせの件の場合の返還請求の法律構成としては、預けたものを返せという契約法上の債務不履行の問題もさることながら、会社が管理を委ねた資金を横領されたことによる損害を賠償させるという不法行為的意味合いが強くなるのではないか。
６  上記を踏まえて、未返却資金を給与から控除することについて考えると、まず、清算事務を履行しない場合は全額を控除するとの事前協定は、賠償予定の禁止に反する可能性がある（労基法16条）。また、未清算分を減給で対処するには、労基法91条の制限がある（一回の制裁事由につき平均賃金の一日分の半額まで、かつ月給総額の１０分の１まで）。
  そこで、毎日清算事務履行の業務命令を出し、それへの違反につき制裁事由一回とカウントして、毎月給の１０分の１ずつをカットしていく方法によれば、適法に回収できるのではないか。
